def asal_mý(x):
    i = 2
    while i*i <= x:
        if x % i == 0:
            return False
        i += 1
    else:
        return True




def asalçarpanlar(N):
    çarpanlar = []  # boþ liste    
    x = 2
    while N > 1:
        # x asal mý?
        asal = True
        i = 2
        while i*i <= x:
            if x % i == 0:
                asal = False
                break
            i += 1

        if asal and N % x == 0:  # x asalsa ve N'yi bölüyorsa...
            çarpanlar.append(x)  # x'i listenin sonuna ekle
            while N % x == 0 :   # N x'e bölünebildiði sürece...
                N = N / x        # N'yi x'e böl, x çarpaný kalmasýn.
        x += 1
    return çarpanlar
